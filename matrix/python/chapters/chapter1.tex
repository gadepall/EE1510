\subsection{Problem}

\begin{problem}
Sketch the vectors
%
	\begin{equation}
	\mbf{a}_1 = \brak{1,1,1}^T , \mbf{a}_2=\brak{0,1,2}^T, \mbf{b}=\brak{6,0,0}^T
	\end{equation}
	%
in the 3-D plane.
\end{problem}
%
\begin{problem}
	Find $x_1, x_2$ such that
	\begin{equation}
x_1 \mbf{a}_1 + x_2 \mbf{a}_2 = \mbf{b}
	\end{equation}
\end{problem}
%
geometrically.
\begin{problem}
Solve the matrix equation 
%
\begin{equation}
\mbf{Ax} = \mbf{b}
\end{equation}
%
where $\mbf{A} = \sbrak{\mbf{a}_1 \,\mbf{a}_2}$ using row reduction.  Comment.
\end{problem}
